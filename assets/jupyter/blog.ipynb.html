palette = sns.color_palette('colorblind')
fig = plt.figure()
ax = fig.add_subplot(projection='polar')
for color in palette:
    rgb_value = mpl.colors.to_rgb(color)
    hsv_value = np.array(mpl.colors.rgb_to_hsv(rgb_value))
    ax.scatter(x=[2 * np.pi * hsv_value[0]], 
               y=[hsv_value[1]],color=color)

ax.set_yticks([])
